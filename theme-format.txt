escapable symbols: ';' '}'
like this: \; \}

better example:

----------------------------------------------------
monitor samsung {
	model SyncMaster 931BW;
	color black;
	native_resolution { w 1440; h 900 }
	escapable_example 1\;2\;3;
	escapable_as_string_literal "1;2;3";
}
----------------------------------------------------

somewhat equals to XML:

----------------------------------------------------
<monitor value="samsung">
	<model value="SyncMaster 931BW"/>
	<color value="black"/>
	<native_resolution>
		<w value="1440"/>
		<h value="900"/>
	</native_resolution>
	<escapable_example value="1;2;3"/>
	<escapable_as_string_literal value="1;2;3"/>
</monitor>
----------------------------------------------------
