PROJECT(BMPANEL C)
SET(BMPANEL_VERSION_MAJOR "2")

    ${CMAKE_CURRENT_SOURCE_DIR}/bmpanel.c

    ${CMAKE_CURRENT_SOURCE_DIR}/message.c 
    ${CMAKE_CURRENT_SOURCE_DIR}/theme-parser.c 
)

ADD_EXECUTABLE("bmpanel${BMPANEL_VERSION_MAJOR}" ${SOURCES})
